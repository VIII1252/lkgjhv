Задача №5: Простое число 
#include <stdio.h>

int main() {
    int number;
    int i;
    int isPrime = 1;
    
    printf("Введите число: ");
    scanf("%d", &number);
    
    for (i = 2; i < number; i++) {
        if (number % i == 0) {
            isPrime = 0;
            break;
        }
    }
    
    if (isPrime == 1) {
        printf("%d - простое число\n", number);
    } else {
        printf("%d - не простое число\n", number);
    }
    
    return 0;
}
