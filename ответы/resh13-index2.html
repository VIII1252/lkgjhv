Задача №13: Функция возведения в степень 
#include 

int power(int base, int exp);

int main() {
    int b, e;

    printf("Введите основание и показатель: ");
    scanf("%d %d", &b, &e);

    printf("%d^%d = %d\n", b, e, power(b, e));
    return 0;
}


int power(int base, int exp) {
    int result = 1;
    for (int i = 0; i < exp; i++) {
        result *= base;
    }
    return result;
}
