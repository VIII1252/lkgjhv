Задача №11: Сумма чисел от 1 до N 
#include 

int main() {
    int n, i = 1, sum = 0;

    printf("Введите N: ");
    scanf("%d", &n);

    while (i <= n) {
        sum += i;  // то же самое, что sum = sum + i;
        i++;
    }

    printf("Сумма от 1 до %d = %d\n", n, sum);
    return 0;
}
