Задача №12: Массивы 
#include 

int main() {
    int arr[5];
    int i;

    // Заполнение
    printf("Введите 5 чисел:\n");
    for (i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }

    // Вывод в обратном порядке
    printf("В обратном порядке:\n");
    for (i = 4; i >= 0; i--) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    return 0;
}
