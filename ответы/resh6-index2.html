Задача №6: Площадь и периметр 
#include <stdio.h>

int main() {
    int length, width;
    
    printf("Введите длину: ");
    scanf("%d", &length);
    
    printf("Введите ширину: ");
    scanf("%d", &width);
    
    printf("Площадь: %d\n", length * width);
    printf("Периметр: %d\n", 2 * (length + width));
    
    return 0;
}
