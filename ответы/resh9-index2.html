Задача №9: Сортировка трех чисел 
#include <stdio.h>

int main() {
    int a, b, c;
    int min, mid, max;
    
    printf("Введите три числа: ");
    scanf("%d %d %d", &a, &b, &c);
    
    min = a;
    if (b < min) min = b;
    if (c < min) min = c;
    
    max = a;
    if (b > max) max = b;
    if (c > max) max = c;
    
    mid = a + b + c - min - max;
    
    printf("%d %d %d\n", min, mid, max);
    
    return 0;
}
