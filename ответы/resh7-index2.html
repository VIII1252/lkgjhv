Задача №7: Положительные и отрицательные 
#include <stdio.h>

int main() {
    int number;
    int positive = 0;
    int negative = 0;
    
    printf("Введите 7 чисел (положительных и отрицательных):\n");
    
    for (int i = 0; i < 7; i++) {
        scanf("%d", &number);
        
        if (number > 0) {
            positive++;
        } else if (number < 0) {
            negative++;
        }
    }
    
    printf("Положительных: %d\n", positive);
    printf("Отрицательных: %d\n", negative);
    
    return 0;
}
