Задача №8: Возведение в степень 
#include <stdio.h>

int main() {
    int a, b, i;
    int result = 1;
    
    printf("Введите число A: ");
    scanf("%d", &a);
    
    printf("Введите степень B: ");
    scanf("%d", &b);
    
    for(i = 1; i <= b; i++) {
        result = result * a;
    }
    
    printf("%d в степени %d = %d\n", a, b, result);
    
    return 0;
}
